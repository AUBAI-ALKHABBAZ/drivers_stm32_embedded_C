
#include "stm32f10x.h"                  // Device header
void delay(int rep);
int main(void)
{
	RCC->APB2ENR |= 0x10; /// Or  0b10000 --> Anabling Preiph GPIOC
	GPIOC->CRH &= 0x0F0FFFFF;  /// Reset the PORT C PIN 13
	GPIOC->CRH |= 0x30300000;  /// Set Port C PIN 13 as Output 
	//GPIOC->CRH |= 0x30300000;  /// Set Port C PIN 13 and 15 as Output 
	//GPIOC->ODR  |= 0x2000; 	/// Set Port C Pin 13 
	GPIOC->ODR  |= 0b1010000000000000; // in binary mode  Set Port C Pin 13 and 15
	//return 0 ;
	while(1){
	GPIOC->ODR  |= 0x2000;     /// on  Set Port C Pin 13 
	delay(10);
	GPIOC->ODR  &= ~0x2000;     /// off  Set Port C Pin 13 	
	delay(10);
	}
	
}
